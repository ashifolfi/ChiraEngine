209 void Device::refreshWindows() {
210  // Render each window
211  for (auto& handle : WINDOWS) {
212  if (!handle)
213  continue;
214 
215  if (Device::isWindowAboutToBeDestroyed(&handle)) {
216  Device::destroyWindow(&handle);
217  continue;
218  }
219 
220  if (!Device::isWindowVisible(&handle)) {
221  handle.layer->update();
222  continue;
223  }
224 
225  SDL_GL_MakeCurrent(handle.window, g_GLContext);
226  glViewport(0, 0, handle.width, handle.height);
227 
228  ImGui::SetCurrentContext(handle.imguiContext);
229  setImGuiConfigPath();
230 
231  Renderer::pushFrameBuffer(*handle.layer->getRawHandle());
232  Renderer::startImGuiFrame(handle.window);
233 
234  handle.layer->update();
235  handle.layer->render();
236 
237  for (auto& [uuid, panel] : handle.panels) {
238  panel->render();
239  }
240 
241  glDisable(GL_DEPTH_TEST);
242 
243  Renderer::endImGuiFrame();
244  Renderer::popFrameBuffer();
245 
246  MeshDataBuilder surface;
247  surface.addSquare({}, {2, -2}, SignedAxis::ZN, 0);
248  surface.setMaterial(Resource::getUniqueUncachedResource<MaterialFrameBuffer>("file://materials/window.json", handle.layer->getRawHandle()).castAssert<IMaterial>());
249  surface.render(glm::identity<glm::mat4>());
250 
251  glEnable(GL_DEPTH_TEST);
252 
253  SDL_GL_SwapWindow(handle.window);
254  }
255 
256  // Process input
257  SDL_Event event;
258  while (SDL_PollEvent(&event)) {
259  auto* handle = reinterpret_cast<Device::WindowHandle*>(SDL_GetWindowData(SDL_GetWindowFromID(event.window.windowID), "handle"));
260  if (!handle)
261  continue;
262 
263  ImGui::SetCurrentContext(handle->imguiContext);
264  ImGui_ImplSDL2_ProcessEvent(&event);
265 
266  switch (event.type) {
267  case SDL_QUIT:
268  for (auto& windowHandle : WINDOWS) {
269  if (windowHandle) {
270  Device::queueDestroyWindow(&windowHandle, true);
271  }
272  }
273  break;
274  case SDL_WINDOWEVENT:
275  switch (event.window.event) {
276  case SDL_WINDOWEVENT_CLOSE:
277  Device::queueDestroyWindow(handle, true);
278  break;
279  case SDL_WINDOWEVENT_RESIZED:
280  case SDL_WINDOWEVENT_SIZE_CHANGED:
281  case SDL_WINDOWEVENT_MAXIMIZED:
282  SDL_GetWindowSizeInPixels(handle->window, &handle->width, &handle->height);
283  handle->layer->setSize({handle->width, handle->height});
284  break;
285  default:
286  // There's quite a few events we don't care about
287  break;
288  }
289  break;
290  case SDL_KEYDOWN:
291  for (const auto& keyEvent : Input::KeyEvent::getEvents()) {
292  if (keyEvent.getEvent() == event.key.keysym.sym && keyEvent.getEventType() == Input::KeyEventType::PRESSED) {
293  keyEvent();
294  }
295  }
296  break;
297  case SDL_KEYUP:
298  for (const auto& keyEvent : Input::KeyEvent::getEvents()) {
299  if (keyEvent.getEvent() == event.key.keysym.sym && keyEvent.getEventType() == Input::KeyEventType::RELEASED) {
300  keyEvent();
301  }
302  }
303  break;
304  case SDL_MOUSEBUTTONDOWN:
305  for (const auto& mouseEvent : Input::MouseEvent::getEvents()) {
306  if (static_cast<uint8_t>(mouseEvent.getEvent()) == event.button.button && mouseEvent.getEventType() == Input::MouseEventType::CLICKED) {
307  mouseEvent(event.button.x, event.button.y, event.button.clicks);
308  }
309  }
310  break;
311  case SDL_MOUSEBUTTONUP:
312  for (const auto& mouseEvent : Input::MouseEvent::getEvents()) {
313  if (static_cast<uint8_t>(mouseEvent.getEvent()) == event.button.button && mouseEvent.getEventType() == Input::MouseEventType::RELEASED) {
314  mouseEvent(event.button.x, event.button.y, event.button.clicks);
315  }
316  }
317  break;
318  case SDL_MOUSEMOTION:
319  for (const auto& mouseMotionEvent : Input::MouseMotionEvent::getEvents()) {
320  if (mouseMotionEvent.getEvent() == Input::MouseMotion::MOVEMENT) {
321  mouseMotionEvent(event.motion.x, event.motion.y, event.motion.xrel, event.motion.yrel);
322  }
323  }
324  break;
325  case SDL_MOUSEWHEEL:
326  for (const auto& mouseMotionEvent : Input::MouseMotionEvent::getEvents()) {
327  if (mouseMotionEvent.getEvent() == Input::MouseMotion::SCROLL) {
328  mouseMotionEvent(event.wheel.x, event.wheel.y, event.wheel.x, event.wheel.y);
329  }
330  }
331  break;
332  default:
333  // todo(input): handle joystick / game controller inputs!
334  break;
335  }
336  }
337 
338  // Handle repeating events
339  // This is a pointer to a static variable in SDL so this is safe
340  static const auto* keyStates = SDL_GetKeyboardState(nullptr);
341  for (const auto& keyEvent : Input::KeyEvent::getEvents()) {
342  if (keyStates[SDL_GetScancodeFromKey(keyEvent.getEvent())] && keyEvent.getEventType() == Input::KeyEventType::REPEATED) {
343  keyEvent();
344  }
345  }
346 }
