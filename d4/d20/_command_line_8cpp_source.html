17 static inline void processConEntry(int& i, int argc, const char* argv[]) {
18  std::string_view conEntry = argv[i];
19  conEntry.remove_prefix(1);
20  if (auto* command = ConEntryRegistry::getConCommand(conEntry)) {
21  i++;
22  std::vector<std::string> input;
23  while (i < argc) {
24  std::string_view current = argv[i];
25  if (current.starts_with('-') || current.starts_with('+')) {
26  break;
27  }
28  input.emplace_back(current.data());
29  }
30  command->fire(input);
31  } else if (auto* var = ConEntryRegistry::getConVar(conEntry)) {
32  i++;
33  std::string value = argv[i];
34  if (value.starts_with('-') || value.starts_with('+')) {
35  LOG_COMMANDLINE.warning("ConVar \"{}\" not given a value!", conEntry);
36  } else {
37  if (var->hasFlag(CON_FLAG_READONLY)) {
38  LOG_COMMANDLINE.warning("ConVar \"{}\" is read-only!", conEntry);
39  } else {
40  var->setValue(value);
41  }
42  i++;
43  }
44  } else {
45  LOG_COMMANDLINE.warning("Unknown console entry \"{}\"", conEntry);
46  i++;
47  }
48 }

59  } else {
60  g_Arguments.push_back(arg);

63  // If it starts with a dash, it's looking for an argument. However, if there's a
64  // dash or plus sign prefixing the next arg there is no argument, so add a blank
65  if (i == argc) {
66  g_Arguments.emplace_back("");
67  } else if (arg.starts_with(OPTION_PREFIX)) {
