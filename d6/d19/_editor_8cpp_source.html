50  Input::KeyEvent::create(Input::Key::SDLK_d, Input::KeyEventType::REPEATED, [&cameraTransform] {

53  });
54  Input::KeyEvent::create(Input::Key::SDLK_SPACE, Input::KeyEventType::REPEATED, [&cameraTransform] {
55  if (Device::isMouseCapturedWindow(Engine::getMainWindow()))
56  cameraTransform.translateWithRotation({0, movementSpeed * (static_cast<float>(Engine::getDeltaTicks()) / 1000), 0});
57  });
58  Input::KeyEvent::create(Input::Key::SDLK_LSHIFT, Input::KeyEventType::REPEATED, [&cameraTransform] {
59  if (Device::isMouseCapturedWindow(Engine::getMainWindow()))
60  cameraTransform.translateWithRotation({0, -movementSpeed * (static_cast<float>(Engine::getDeltaTicks()) / 1000), 0});
61  });
62  Input::MouseEvent::create(Input::Mouse::BUTTON_RIGHT, Input::MouseEventType::CLICKED, [](int, int, uint8_t) {
63  Device::setMouseCapturedWindow(Engine::getMainWindow(), true);
64  });
65  Input::MouseEvent::create(Input::Mouse::BUTTON_RIGHT, Input::MouseEventType::RELEASED, [](int, int, uint8_t) {
66  Device::setMouseCapturedWindow(Engine::getMainWindow(), false);
67  });
68  Input::MouseMotionEvent::create(Input::MouseMotion::MOVEMENT, Input::MouseMotionEventType::NOT_APPLICABLE, [&cameraTransform](int, int, int xRel, int yRel) {
69  if (!Device::isMouseCapturedWindow(Engine::getMainWindow()))
70  return;
71 
72  float xOffset = static_cast<float>(xRel) * mouseSensitivity * (static_cast<float>(Engine::getDeltaTicks()) / 1000);
73  float yOffset = static_cast<float>(yRel) * mouseSensitivity * (static_cast<float>(Engine::getDeltaTicks()) / 1000);
74 
75  float yaw = glm::degrees(cameraTransform.getYaw());
76  yaw -= xOffset;
77 
78  float pitch = glm::degrees(cameraTransform.getPitch());
79  pitch -= yOffset;
80  pitch = std::clamp(pitch, -89.9f, 89.9f);
81 
82  cameraTransform.setYaw(glm::radians(yaw));
83  cameraTransform.setPitch(glm::radians(pitch));
84  });
85  Input::MouseMotionEvent::create(Input::MouseMotion::SCROLL, Input::MouseMotionEventType::NOT_APPLICABLE, [&cameraTransform](int x, int y, int, int) {
86  if (Device::isMouseCapturedWindow(Engine::getMainWindow()))
87  cameraTransform.translateWithRotation({
88  static_cast<float>(x) * scrollModifier / 40,
89  0,
90  -static_cast<float>(y) * scrollModifier / 40 // negate for OpenGL
91  });
92  });

239  void setLoadedFile(const std::string& meshName) {
240  if (scene->hasEntity(this->meshId) && meshName == scene->getEntity(this->meshId)->getComponent<MeshComponent>().getMeshResource()->getIdentifier())
241  return;
242  if (!Resource::hasResource(meshName)) {
243  Device::popupError(TRF("error.modelviewer.resource_is_invalid", meshName));
244  return;
245  }
246  if (!scene->hasEntity(this->meshId)) {
247  auto* entity = scene->addEntity();
248  this->meshId = entity->getUUID();
249  }
250  auto* entity = scene->getEntity(this->meshId);
251  entity->tryRemoveComponent<MeshComponent>();
252  entity->addComponent<MeshComponent>(meshName);
253  this->loadedFile = meshName;
254  }
