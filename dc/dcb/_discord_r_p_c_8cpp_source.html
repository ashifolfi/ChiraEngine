104 void DiscordRPC::updatePresence() {
105  if (DiscordRPC::isModifiedSinceLastUpdate && discord_enable.getValue<bool>()) {
106  DiscordRichPresence discordPresence;
107  memset(&discordPresence, 0, sizeof(discordPresence));
108  if (!DiscordRPC::state.empty())
109  discordPresence.state = DiscordRPC::state.c_str();
110  if (!DiscordRPC::details.empty())
111  discordPresence.details = DiscordRPC::details.c_str();
112  if (DiscordRPC::startTimestamp >= 0)
113  discordPresence.startTimestamp = DiscordRPC::startTimestamp;
114  if (DiscordRPC::endTimestamp >= 0)
115  discordPresence.endTimestamp = DiscordRPC::endTimestamp;
116  if (!DiscordRPC::largeImage.empty())
117  discordPresence.largeImageKey = DiscordRPC::largeImage.c_str();
118  if (!DiscordRPC::largeImageText.empty())
119  discordPresence.largeImageText = DiscordRPC::largeImageText.c_str();
120  if (!DiscordRPC::smallImage.empty())
121  discordPresence.smallImageKey = DiscordRPC::smallImage.c_str();
122  if (!DiscordRPC::smallImageText.empty())
123  discordPresence.smallImageText = DiscordRPC::smallImageText.c_str();
124 
125  DiscordButton buttons[2] {{}, {}};
126  if (!DiscordRPC::button1.url.empty()) {
127  buttons[0].label = DiscordRPC::button1.name.c_str();
128  buttons[0].url = DiscordRPC::button1.url.c_str();
129  }
130  if (!DiscordRPC::button2.url.empty()) {
131  buttons[1].label = DiscordRPC::button2.name.c_str();
132  buttons[1].url = DiscordRPC::button2.url.c_str();
133  }
134  discordPresence.buttons = buttons;
135 
136  discordPresence.instance = 0;
137  Discord_UpdatePresence(&discordPresence);
138  DiscordRPC::isModifiedSinceLastUpdate = false;
139  }
